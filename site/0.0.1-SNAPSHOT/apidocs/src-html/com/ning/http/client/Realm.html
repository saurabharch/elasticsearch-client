197    @Override
198    public boolean equals(Object o) {
199        if (this == o) return true;
200        if (o == null || getClass() != o.getClass()) return false;
201
202        Realm realm = (Realm) o;
203
204        if (algorithm != null ? !algorithm.equals(realm.algorithm) : realm.algorithm != null) return false;
205        if (cnonce != null ? !cnonce.equals(realm.cnonce) : realm.cnonce != null) return false;
206        if (nc != null ? !nc.equals(realm.nc) : realm.nc != null) return false;
207        if (nonce != null ? !nonce.equals(realm.nonce) : realm.nonce != null) return false;
208        if (password != null ? !password.equals(realm.password) : realm.password != null) return false;
209        if (principal != null ? !principal.equals(realm.principal) : realm.principal != null) return false;
210        if (qop != null ? !qop.equals(realm.qop) : realm.qop != null) return false;
211        if (realmName != null ? !realmName.equals(realm.realmName) : realm.realmName != null) return false;
212        if (response != null ? !response.equals(realm.response) : realm.response != null) return false;
213        if (scheme != realm.scheme) return false;
214        if (uri != null ? !uri.equals(realm.uri) : realm.uri != null) return false;
215
216        return true;
217    }

237    @Override
238    public int hashCode() {
239        int result = principal != null ? principal.hashCode() : 0;
240        result = 31 * result + (password != null ? password.hashCode() : 0);
241        result = 31 * result + (scheme != null ? scheme.hashCode() : 0);
242        result = 31 * result + (realmName != null ? realmName.hashCode() : 0);
243        result = 31 * result + (nonce != null ? nonce.hashCode() : 0);
244        result = 31 * result + (algorithm != null ? algorithm.hashCode() : 0);
245        result = 31 * result + (response != null ? response.hashCode() : 0);
246        result = 31 * result + (qop != null ? qop.hashCode() : 0);
247        result = 31 * result + (nc != null ? nc.hashCode() : 0);
248        result = 31 * result + (cnonce != null ? cnonce.hashCode() : 0);
249        result = 31 * result + (uri != null ? uri.hashCode() : 0);
250        return result;
251    }

428        public RealmBuilder parseWWWAuthenticateHeader(String headerLine) {
429            setRealmName(match(headerLine, "realm"));
430            setNonce(match(headerLine, "nonce"));
431            setAlgorithm(match(headerLine, "algorithm"));
432            setOpaque(match(headerLine, "opaque"));
433            setQop(match(headerLine, "qop"));
434            if (getNonce() != null && !getNonce().equalsIgnoreCase("")) {
435                setScheme(AuthScheme.DIGEST);
436            } else {
437                setScheme(AuthScheme.BASIC);
438            }
439            return this;
440        }

467        private void newCnonce() {
468            try {
469                MessageDigest md = MessageDigest.getInstance("MD5");
470                byte[] b = md.digest(String.valueOf(System.currentTimeMillis()).getBytes("ISO-8859-1"));
471                cnonce = toHexString(b);
472            } catch (Exception e) {
473                throw new SecurityException(e);
474            }
475        }
476
477        /**
478         * TODO: A Pattern/Matcher may be better.
479         */
480        private String match(String headerLine, String token) {
481            if (headerLine == null) {
482                return "";
483            }
484
485            int match = headerLine.indexOf(token);
486            if (match <= 0) return "";
487
488            // = to skip
489            match += token.length() + 1;
490            int traillingComa = headerLine.indexOf(",", match);
491            String value = headerLine.substring(match, traillingComa > 0 ? traillingComa : headerLine.length());
492            value = value.endsWith("\"") ? value.substring(0, value.length() - 1) : value;
493            return value.startsWith("\"") ? value.substring(1) : value;
494        }

505        private void newResponse() throws UnsupportedEncodingException {
506            MessageDigest md = null;
507            try {
508                md = MessageDigest.getInstance("MD5");
509            } catch (NoSuchAlgorithmException e) {
510                throw new SecurityException(e);
511            }
512            md.update(new StringBuilder(principal)
513                    .append(":")
514                    .append(realmName)
515                    .append(":")
516                    .append(password)
517                    .toString().getBytes("ISO-8859-1"));
518            byte[] ha1 = md.digest();
519
520            md.reset();
521
522            //HA2 if qop is auth-int is methodName:url:md5(entityBody)
523            md.update(new StringBuilder(methodName)
524                    .append(':')
525                    .append(uri).toString().getBytes("ISO-8859-1"));
526            byte[] ha2 = md.digest();
527
528            if(qop==null || qop.equals("")) {
529                 md.update(new StringBuilder(toBase16(ha1))
530                    .append(':')
531                    .append(nonce)
532                    .append(':')
533                    .append(toBase16(ha2)).toString().getBytes("ISO-8859-1"));
534
535             } else {
536                 //qop ="auth" or "auth-int"
537                 md.update(new StringBuilder(toBase16(ha1))
538                    .append(':')
539                    .append(nonce)
540                    .append(':')
541                    .append(NC)
542                    .append(':')
543                    .append(cnonce)
544                    .append(':')
545                    .append(qop)
546                    .append(':')
547                    .append(toBase16(ha2)).toString().getBytes("ISO-8859-1"));
548            }
549
550            byte[] digest = md.digest();
551
552            response = toHexString(digest);
553        }
554
555        private static String toHexString(byte[] data) {
556            StringBuilder buffer = new StringBuilder();
557            for (int i = 0; i < data.length; i++) {
558                buffer.append(Integer.toHexString((data[i] & 0xf0) >>> 4));
559                buffer.append(Integer.toHexString(data[i] & 0x0f));
560            }
561            return buffer.toString();
562        }
563
564        private static String toBase16(byte[] bytes) {
565            int base = 16;
566            StringBuilder buf = new StringBuilder();
567            for (byte b : bytes) {
568                int bi = 0xff & b;
569                int c = '0' + (bi / base) % base;
570                if (c > '9')
571                    c = 'a' + (c - '0' - 10);
572                buf.append((char) c);
573                c = '0' + bi % base;
574                if (c > '9')
575                    c = 'a' + (c - '0' - 10);
576                buf.append((char) c);
577            }
578            return buf.toString();
579        }

581        /**
582         * Build a {@link Realm}
583         *
584         * @return a {@link Realm}
585         */
586        public Realm build() {
587
588            // Avoid generating
589            if (nonce != null && !nonce.equals("")) {
590                newCnonce();
591                try {
592                    newResponse();
593                } catch (UnsupportedEncodingException e) {
594                    throw new RuntimeException(e);
595                }
596            }
597
598            return new Realm(scheme,
599                    principal,
600                    password,
601                    realmName,
602                    nonce,
603                    algorithm,
604                    response,
605                    qop,
606                    nc,
607                    cnonce,
608                    uri,
609                    methodName,
610                    usePreemptive,
611                    domain,
612                    enc,
613                    host,
614                    messageType2Received,
615                    opaque);
616        }
