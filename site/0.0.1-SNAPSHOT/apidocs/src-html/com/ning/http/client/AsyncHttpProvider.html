016package com.ning.http.client;
017
018import java.io.IOException;
019import java.util.List;
020
021/**
022 * Interface to be used when implementing custom asynchronous I/O HTTP client.
023 * By default, the {@link com.ning.http.client.providers.netty.NettyAsyncHttpProvider} is used.
024 */
025public interface AsyncHttpProvider {
026
027    /**
028     * Execute the request and invoke the {@link AsyncHandler} when the response arrive.
029     *
030     * @param handler an instance of {@link AsyncHandler}
031     * @return a {@link ListenableFuture} of Type T.
032     * @throws IOException
033     */
034    public <T> ListenableFuture<T> execute(Request request, AsyncHandler<T> handler) throws IOException;
035
036    /**
037     * Close the current underlying TCP/HTTP connection.
038     */
039    public void close();
040
041    /**
042     * Prepare a {@link Response}
043     *
044     * @param status    {@link HttpResponseStatus}
045     * @param headers   {@link HttpResponseHeaders}
046     * @param bodyParts list of {@link HttpResponseBodyPart}
047     * @return a {@link Response}
048     */
049    public Response prepareResponse(HttpResponseStatus status,
050                                    HttpResponseHeaders headers,
051                                    List<HttpResponseBodyPart> bodyParts);
052}
