022/**
023 * Represents char[], as a slice (offset + length) into an existing char[].
024 * The {@link #chars} member should never be null; use
025 * {@link #EMPTY_CHARS} if necessary.
026 */
027public final class CharsRef implements Comparable<CharsRef>, CharSequence, Cloneable {
028  public static final char[] EMPTY_CHARS = new char[0];
029  public char[] chars;
030  public int offset;
031  public int length;
032
033  /**
034   * Creates a new {@link CharsRef} initialized an empty array zero-length
035   */
036  public CharsRef() {
037    this(EMPTY_CHARS, 0, 0);
038  }
039
040  /**
041   * Creates a new {@link CharsRef} initialized with an array of the given
042   * capacity
043   */
044  public CharsRef(int capacity) {
045    chars = new char[capacity];
046  }
047
048  /**
049   * Creates a new {@link CharsRef} initialized with the given array, offset and
050   * length
051   */
052  public CharsRef(char[] chars, int offset, int length) {
053    assert chars != null;
054    assert chars.length >= offset + length;
055    this.chars = chars;
056    this.offset = offset;
057    this.length = length;
058  }
059
060  /**
061   * Creates a new {@link CharsRef} initialized with the given Strings character
062   * array
063   */
064  public CharsRef(String string) {
065    this.chars = string.toCharArray();
066    this.offset = 0;
067    this.length = chars.length;
068  }
069
070  @Override
071  public CharsRef clone() {
072    return new CharsRef(chars, offset, length);
073  }
074
075  @Override
076  public int hashCode() {
077    final int prime = 31;
078    int result = 0;
079    final int end = offset + length;
080    for (int i = offset; i < end; i++) {
081      result = prime * result + chars[i];
082    }
083    return result;
084  }
085
086  @Override
087  public boolean equals(Object other) {
088    if (other == null) {
089      return false;
090    }
091    if (other instanceof CharsRef) {
092      return this.charsEquals((CharsRef) other);
093    }
094    return false;
095  }
096
097  public boolean charsEquals(CharsRef other) {
098    if (length == other.length) {
099      int otherUpto = other.offset;
100      final char[] otherChars = other.chars;
101      final int end = offset + length;
102      for (int upto = offset; upto < end; upto++, otherUpto++) {
103        if (chars[upto] != otherChars[otherUpto]) {
104          return false;
105        }
106      }
107      return true;
108    } else {
109      return false;
110    }
111  }
112
113  /** Signed int order comparison */
114  public int compareTo(CharsRef other) {
115    if (this == other)
116      return 0;
117
118    final char[] aChars = this.chars;
119    int aUpto = this.offset;
120    final char[] bChars = other.chars;
121    int bUpto = other.offset;
122
123    final int aStop = aUpto + Math.min(this.length, other.length);
124
125    while (aUpto < aStop) {
126      int aInt = aChars[aUpto++];
127      int bInt = bChars[bUpto++];
128      if (aInt > bInt) {
129        return 1;
130      } else if (aInt < bInt) {
131        return -1;
132      }
133    }
134
135    // One is a prefix of the other, or, they are equal:
136    return this.length - other.length;
137  }
138  
139  /**
140   * Copies the given {@link CharsRef} referenced content into this instance
141   * starting at offset 0.
142   * 
143   * @param other
144   *          the {@link CharsRef} to copy
145   */
146  // TODO: why does this behave differently/not invoke copyChars(char[], int, int) ???
147  public void copyChars(CharsRef other) {
148    if (chars == null) {
149      chars = new char[other.length];
150    } else {
151      chars = ArrayUtil.grow(chars, other.length);
152    }
153    System.arraycopy(other.chars, other.offset, chars, 0, other.length);
154    length = other.length;
155    offset = 0;
156  }
157
158  public void grow(int newLength) {
159    if (chars.length < newLength) {
160      chars = ArrayUtil.grow(chars, newLength);
161    }
162  }
163
164  /**
165   * Copies the given array into this CharsRef starting at offset 0
166   */
167  public void copyChars(char[] otherChars, int otherOffset, int otherLength) {
168    grow(otherLength);
169    System.arraycopy(otherChars, otherOffset, this.chars, 0,
170        otherLength);
171    this.offset = 0;
172    this.length = otherLength;
173  }
174
175  /**
176   * Appends the given array to this CharsRef
177   */
178  public void append(char[] otherChars, int otherOffset, int otherLength) {
179    final int newLength = length + otherLength;
180    grow(this.offset + newLength);
181    System.arraycopy(otherChars, otherOffset, this.chars, this.offset+length,
182        otherLength);
183    this.length += otherLength;
184  }
185
186  @Override
187  public String toString() {
188    return new String(chars, offset, length);
189  }
190
191  public int length() {
192    return length;
193  }
194
195  public char charAt(int index) {
196    return chars[offset + index];
197  }
198
199  public CharSequence subSequence(int start, int end) {
200    return new CharsRef(chars, offset + start, offset + end - 1);
201  }
202  
203  private final static Comparator<CharsRef> utf16SortedAsUTF8SortOrder = new UTF16SortedAsUTF8Comparator();
204  
205  public static Comparator<CharsRef> getUTF16SortedAsUTF8Comparator() {
206    return utf16SortedAsUTF8SortOrder;
207  }
208  
209  private static class UTF16SortedAsUTF8Comparator implements Comparator<CharsRef> {
210    // Only singleton
211    private UTF16SortedAsUTF8Comparator() {};
212
213    public int compare(CharsRef a, CharsRef b) {
214      if (a == b)
215        return 0;
216
217      final char[] aChars = a.chars;
218      int aUpto = a.offset;
219      final char[] bChars = b.chars;
220      int bUpto = b.offset;
221
222      final int aStop = aUpto + Math.min(a.length, b.length);
223
224      while (aUpto < aStop) {
225        char aChar = aChars[aUpto++];
226        char bChar = bChars[bUpto++];
227        if (aChar != bChar) {
228          // http://icu-project.org/docs/papers/utf16_code_point_order.html
229          
230          /* aChar != bChar, fix up each one if they're both in or above the surrogate range, then compare them */
231          if (aChar >= 0xd800 && bChar >= 0xd800) {
232            if (aChar >= 0xe000) {
233              aChar -= 0x800;
234            } else {
235              aChar += 0x2000;
236            }
237            
238            if (bChar >= 0xe000) {
239              bChar -= 0x800;
240            } else {
241              bChar += 0x2000;
242            }
243          }
244          
245          /* now aChar and bChar are in code point order */
246          return (int)aChar - (int)bChar; /* int must be 32 bits wide */
247        }
248      }
249
250      // One is a prefix of the other, or, they are equal:
251      return a.length - b.length;
252    }
253  }
254  
255  /**
256   * Creates a new CharsRef that points to a copy of the chars from 
257   * <code>other</code>
258   * <p>
259   * The returned CharsRef will have a length of other.length
260   * and an offset of zero.
261   */
262  public static CharsRef deepCopyOf(CharsRef other) {
263    CharsRef clone = new CharsRef();
264    clone.copyChars(other);
265    return clone;
266  }
