007package org.elasticsearch.common.util.concurrent.jsr166y;
008
009import java.util.Random;
010
011/**
012 * A random number generator isolated to the current thread.  Like the
013 * global {@link java.util.Random} generator used by the {@link
014 * java.lang.Math} class, a {@code ThreadLocalRandom} is initialized
015 * with an internally generated seed that may not otherwise be
016 * modified. When applicable, use of {@code ThreadLocalRandom} rather
017 * than shared {@code Random} objects in concurrent programs will
018 * typically encounter much less overhead and contention.  Use of
019 * {@code ThreadLocalRandom} is particularly appropriate when multiple
020 * tasks (for example, each a {@link ForkJoinTask}) use random numbers
021 * in parallel in thread pools.
022 *
023 * <p>Usages of this class should typically be of the form:
024 * {@code ThreadLocalRandom.current().nextX(...)} (where
025 * {@code X} is {@code Int}, {@code Long}, etc).
026 * When all usages are of this form, it is never possible to
027 * accidently share a {@code ThreadLocalRandom} across multiple threads.
028 *
029 * <p>This class also provides additional commonly used bounded random
030 * generation methods.
031 *
032 * @since 1.7
033 * @author Doug Lea
034 */
035public class ThreadLocalRandom extends Random {
036    // same constants as Random, but must be redeclared because private
037    private static final long multiplier = 0x5DEECE66DL;
038    private static final long addend = 0xBL;
039    private static final long mask = (1L << 48) - 1;
040
041    /**
042     * The random seed. We can't use super.seed.
043     */
044    private long rnd;
045
046    /**
047     * Initialization flag to permit calls to setSeed to succeed only
048     * while executing the Random constructor.  We can't allow others
049     * since it would cause setting seed in one part of a program to
050     * unintentionally impact other usages by the thread.
051     */
052    boolean initialized;
053
054    // Padding to help avoid memory contention among seed updates in
055    // different TLRs in the common case that they are located near
056    // each other.
057    private long pad0, pad1, pad2, pad3, pad4, pad5, pad6, pad7;
058
059    /**
060     * The actual ThreadLocal
061     */
062    private static final ThreadLocal<ThreadLocalRandom> localRandom =
063        new ThreadLocal<ThreadLocalRandom>() {
064            protected ThreadLocalRandom initialValue() {
065                return new ThreadLocalRandom();
066            }
067    };
068
069
070    /**
071     * Constructor called only by localRandom.initialValue.
072     */
073    ThreadLocalRandom() {
074        super();
075        initialized = true;
076    }
077
078    /**
079     * Returns the current thread's {@code ThreadLocalRandom}.
080     *
081     * @return the current thread's {@code ThreadLocalRandom}
082     */
083    public static ThreadLocalRandom current() {
084        return localRandom.get();
085    }
086
087    /**
088     * Throws {@code UnsupportedOperationException}.  Setting seeds in
089     * this generator is not supported.
090     *
091     * @throws UnsupportedOperationException always
092     */
093    public void setSeed(long seed) {
094        if (initialized)
095            throw new UnsupportedOperationException();
096        rnd = (seed ^ multiplier) & mask;
097    }
098
099    protected int next(int bits) {
100        rnd = (rnd * multiplier + addend) & mask;
101        return (int) (rnd >>> (48-bits));
102    }
103
104    /**
105     * Returns a pseudorandom, uniformly distributed value between the
106     * given least value (inclusive) and bound (exclusive).
107     *
108     * @param least the least value returned
109     * @param bound the upper bound (exclusive)
110     * @throws IllegalArgumentException if least greater than or equal
111     * to bound
112     * @return the next value
113     */
114    public int nextInt(int least, int bound) {
115        if (least >= bound)
116            throw new IllegalArgumentException();
117        return nextInt(bound - least) + least;
118    }
119
120    /**
121     * Returns a pseudorandom, uniformly distributed value
122     * between 0 (inclusive) and the specified value (exclusive).
123     *
124     * @param n the bound on the random number to be returned.  Must be
125     *        positive.
126     * @return the next value
127     * @throws IllegalArgumentException if n is not positive
128     */
129    public long nextLong(long n) {
130        if (n <= 0)
131            throw new IllegalArgumentException("n must be positive");
132        // Divide n by two until small enough for nextInt. On each
133        // iteration (at most 31 of them but usually much less),
134        // randomly choose both whether to include high bit in result
135        // (offset) and whether to continue with the lower vs upper
136        // half (which makes a difference only if odd).
137        long offset = 0;
138        while (n >= Integer.MAX_VALUE) {
139            int bits = next(2);
140            long half = n >>> 1;
141            long nextn = ((bits & 2) == 0) ? half : n - half;
142            if ((bits & 1) == 0)
143                offset += n - nextn;
144            n = nextn;
145        }
146        return offset + nextInt((int) n);
147    }
148
149    /**
150     * Returns a pseudorandom, uniformly distributed value between the
151     * given least value (inclusive) and bound (exclusive).
152     *
153     * @param least the least value returned
154     * @param bound the upper bound (exclusive)
155     * @return the next value
156     * @throws IllegalArgumentException if least greater than or equal
157     * to bound
158     */
159    public long nextLong(long least, long bound) {
160        if (least >= bound)
161            throw new IllegalArgumentException();
162        return nextLong(bound - least) + least;
163    }
164
165    /**
166     * Returns a pseudorandom, uniformly distributed {@code double} value
167     * between 0 (inclusive) and the specified value (exclusive).
168     *
169     * @param n the bound on the random number to be returned.  Must be
170     *        positive.
171     * @return the next value
172     * @throws IllegalArgumentException if n is not positive
173     */
174    public double nextDouble(double n) {
175        if (n <= 0)
176            throw new IllegalArgumentException("n must be positive");
177        return nextDouble() * n;
178    }
179
180    /**
181     * Returns a pseudorandom, uniformly distributed value between the
182     * given least value (inclusive) and bound (exclusive).
183     *
184     * @param least the least value returned
185     * @param bound the upper bound (exclusive)
186     * @return the next value
187     * @throws IllegalArgumentException if least greater than or equal
188     * to bound
189     */
190    public double nextDouble(double least, double bound) {
191        if (least >= bound)
192            throw new IllegalArgumentException();
193        return nextDouble() * (bound - least) + least;
194    }
195
196    private static final long serialVersionUID = -5851777807851030925L;
197}
