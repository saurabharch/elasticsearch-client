007package org.elasticsearch.common.util.concurrent.jsr166y;
008import java.util.concurrent.*;
009
010/**
011 * A {@link BlockingQueue} in which producers may wait for consumers
012 * to receive elements.  A {@code TransferQueue} may be useful for
013 * example in message passing applications in which producers
014 * sometimes (using method {@link #transfer}) await receipt of
015 * elements by consumers invoking {@code take} or {@code poll}, while
016 * at other times enqueue elements (via method {@code put}) without
017 * waiting for receipt.
018 * {@linkplain #tryTransfer(Object) Non-blocking} and
019 * {@linkplain #tryTransfer(Object,long,TimeUnit) time-out} versions of
020 * {@code tryTransfer} are also available.
021 * A {@code TransferQueue} may also be queried, via {@link
022 * #hasWaitingConsumer}, whether there are any threads waiting for
023 * items, which is a converse analogy to a {@code peek} operation.
024 *
025 * <p>Like other blocking queues, a {@code TransferQueue} may be
026 * capacity bounded.  If so, an attempted transfer operation may
027 * initially block waiting for available space, and/or subsequently
028 * block waiting for reception by a consumer.  Note that in a queue
029 * with zero capacity, such as {@link SynchronousQueue}, {@code put}
030 * and {@code transfer} are effectively synonymous.
031 *
032 * <p>This interface is a member of the
033 * <a href="{@docRoot}/../technotes/guides/collections/index.html">
034 * Java Collections Framework</a>.
035 *
036 * @since 1.7
037 * @author Doug Lea
038 * @param <E> the type of elements held in this collection
039 */
040public interface TransferQueue<E> extends BlockingQueue<E> {
041    /**
042     * Transfers the element to a waiting consumer immediately, if possible.
043     *
044     * <p>More precisely, transfers the specified element immediately
045     * if there exists a consumer already waiting to receive it (in
046     * {@link #take} or timed {@link #poll(long,TimeUnit) poll}),
047     * otherwise returning {@code false} without enqueuing the element.
048     *
049     * @param e the element to transfer
050     * @return {@code true} if the element was transferred, else
051     *         {@code false}
052     * @throws ClassCastException if the class of the specified element
053     *         prevents it from being added to this queue
054     * @throws NullPointerException if the specified element is null
055     * @throws IllegalArgumentException if some property of the specified
056     *         element prevents it from being added to this queue
057     */
058    boolean tryTransfer(E e);
059
060    /**
061     * Transfers the element to a consumer, waiting if necessary to do so.
062     *
063     * <p>More precisely, transfers the specified element immediately
064     * if there exists a consumer already waiting to receive it (in
065     * {@link #take} or timed {@link #poll(long,TimeUnit) poll}),
066     * else waits until the element is received by a consumer.
067     *
068     * @param e the element to transfer
069     * @throws InterruptedException if interrupted while waiting,
070     *         in which case the element is not left enqueued
071     * @throws ClassCastException if the class of the specified element
072     *         prevents it from being added to this queue
073     * @throws NullPointerException if the specified element is null
074     * @throws IllegalArgumentException if some property of the specified
075     *         element prevents it from being added to this queue
076     */
077    void transfer(E e) throws InterruptedException;
078
079    /**
080     * Transfers the element to a consumer if it is possible to do so
081     * before the timeout elapses.
082     *
083     * <p>More precisely, transfers the specified element immediately
084     * if there exists a consumer already waiting to receive it (in
085     * {@link #take} or timed {@link #poll(long,TimeUnit) poll}),
086     * else waits until the element is received by a consumer,
087     * returning {@code false} if the specified wait time elapses
088     * before the element can be transferred.
089     *
090     * @param e the element to transfer
091     * @param timeout how long to wait before giving up, in units of
092     *        {@code unit}
093     * @param unit a {@code TimeUnit} determining how to interpret the
094     *        {@code timeout} parameter
095     * @return {@code true} if successful, or {@code false} if
096     *         the specified waiting time elapses before completion,
097     *         in which case the element is not left enqueued
098     * @throws InterruptedException if interrupted while waiting,
099     *         in which case the element is not left enqueued
100     * @throws ClassCastException if the class of the specified element
101     *         prevents it from being added to this queue
102     * @throws NullPointerException if the specified element is null
103     * @throws IllegalArgumentException if some property of the specified
104     *         element prevents it from being added to this queue
105     */
106    boolean tryTransfer(E e, long timeout, TimeUnit unit)
107        throws InterruptedException;
108
109    /**
110     * Returns {@code true} if there is at least one consumer waiting
111     * to receive an element via {@link #take} or
112     * timed {@link #poll(long,TimeUnit) poll}.
113     * The return value represents a momentary state of affairs.
114     *
115     * @return {@code true} if there is at least one waiting consumer
116     */
117    boolean hasWaitingConsumer();
118
119    /**
120     * Returns an estimate of the number of consumers waiting to
121     * receive elements via {@link #take} or timed
122     * {@link #poll(long,TimeUnit) poll}.  The return value is an
123     * approximation of a momentary state of affairs, that may be
124     * inaccurate if consumers have completed or given up waiting.
125     * The value may be useful for monitoring and heuristics, but
126     * not for synchronization control.  Implementations of this
127     * method are likely to be noticeably slower than those for
128     * {@link #hasWaitingConsumer}.
129     *
130     * @return the number of consumers waiting to receive elements
131     */
132    int getWaitingConsumerCount();
133}
